153  public static Header ParseFromFile(string fn) {
154  Header h = new Header();
155  MessageLog.Add(String.Format("Parsing header {0}.", fn));
156  string fl = string.Join("\r\n", File.ReadAllLines(fn)).ToUpper();
157 
158  // We temporarily assume `en-US` locale so that string-to-double conversion
159  // always uses '.' as the decimal separator.
160  CultureInfo ci = Thread.CurrentThread.CurrentCulture;
161  Thread.CurrentThread.CurrentCulture = CultureInfo.GetCultureInfoByIetfLanguageTag("en-US");
162 
163  try {
164  // Loop through possible fields looking for them in the header file.
165  foreach (Field f in Fields) {
166  Match m = Regex.Match(fl, string.Format("{0}\\s+(.+)", f.Name.ToUpper()));
167 
168  // Check if there's a match and also validate it if applicable.
169  if (m.Success &&
170  ((f.Validator == null) ||
171  Regex.Match(
172  m.Groups[1].Value,
173  String.Format("^{0}(?:\n|\r\n?)", f.Validator),
174  RegexOptions.Multiline
175  ).Success)
176  ) {
177  // Perform conversion to the field's target type.
178  TypeConverter tc = TypeDescriptor.GetConverter(f.Type);
179  f.StringValue = m.Groups[1].Value;
180  typeof(Header).GetField(f.Name).SetValue(h, tc.ConvertFromString(f.StringValue));
181  } else if (f.DefaultValue != null) {
182  // If a default value is present but either the field is missing
183  // from the header or validation failed, use the default but emit a warning.
184  MessageLog.Add(
185  String.Format(
186  "Value of header attribute {0} is missing or invalid, using default of \"{1}\".",
187  f.Name, f.DefaultValue
188  ),
189  MessageLog.EventType.WARN
190  );
191 
192  typeof(Header).GetField(f.Name).SetValue(h, f.DefaultValue);
193  f.StringValue = f.DefaultValue.ToString();
194  } else {
195  // We fail here because the field is required (no default value), but
196  // is missing from the header file or its validation failed.
197  throw new ParseException(string.Format("Missing required header attribute {0}.", f.Name));
198  }
199  }
200  } catch (Exception ex) {
201  // Bail out, but first convert to a ParseException.
202  if (ex is ParseException) {
203  throw;
204  } else {
205  throw new ParseException(ex.Message);
206  }
207  } finally {
208  // Reset locale settings.
209  Thread.CurrentThread.CurrentCulture = ci;
210 
211  // TODO: Compute other field values.
212  }
213  return h;
214  }
215  }

396  private static void parseAsync(object sender, DoWorkEventArgs e) {
397  BackgroundWorker worker = sender as BackgroundWorker;
398  Dataset instance = new Dataset(e.Argument as string);
399  MessageLog.Add(String.Format("Starting async parse of {0} bands of binary data, pixel dimensions are {1}×{2}.",
400  instance.Header.NBands, instance.Header.NCols, instance.Header.NRows));
401 
402  // Start timekeeping.
403  Stopwatch st = new Stopwatch();
404  st.Start();
405 
406  try {
407  // Create a FileStream that is optimized for sequential scanning.
408  using (FileStream fs = new FileStream(
409  instance.FileName, FileMode.Open, FileAccess.Read, FileShare.Read,
410  0x100000, FileOptions.SequentialScan
411  )) {
412  BitmapData[] bd = (from RasterBand b in instance.Bands select b.BitmapData).ToArray();
413  int dataSize = (int)(instance.Header.NRows * instance.Header.NCols * instance.Header.NBands);
414 
415  byte[] buffer = new byte[dataSize];
416  fs.Read(buffer, 0, (int)(dataSize));
417 
418  unsafe {
419  // We read into unmanaged memory streams, as this seems to be faster than array operations.
420  UnmanagedMemoryStream[] ums = new UnmanagedMemoryStream[instance.Header.NBands];
421 
422  for (int i = 0; i < instance.Header.NBands; i++) {
423  ums[i] = new UnmanagedMemoryStream((byte *) bd[i].Scan0, dataSize, dataSize, FileAccess.Write);
424  }
425 
426  // We read data based on the file layout:
427  // BIL: rows -> bands -> a row of a band at a time
428  // BIP: rows -> cols -> bands -> a single pixel at a time
429  // BSQ: bands -> a whole band at a time
430 
431  // We also check if the user has cancelled the operation and break accordingly.
432  switch (instance.Header.Layout) {
433  case Header.LayoutEnum.BIL:
434  for (int y = 0; y < instance.Header.NRows; y++) {
435  for (int b = 0; b < instance.Header.NBands; b++) {
436  ums[b].Write(buffer,
437  (int) ((y * instance.Header.NBands + b) * instance.Header.NCols),
438  (int) instance.Header.NCols
439  );
440  }
441 
442  if (worker.CancellationPending) {
443  e.Cancel = true;
444  goto cleanup;
445  }
446 
447  worker.ReportProgress((int) (((double) y / instance.Header.NRows) * 100));
448  }
449  break;
450 
451  case Header.LayoutEnum.BIP:
452  for (int y = 0; y < instance.Header.NRows; y++) {
453  for (int x = 0; x < instance.Header.NCols; x++) {
454  for (int b = 0; b < instance.Header.NBands; b++) {
455  ums[b].Write(buffer,
456  (int) ((x + y * instance.Header.NCols) * instance.Header.NBands + b),
457  1
458  );
459  }
460  }
461 
462  if (worker.CancellationPending) {
463  e.Cancel = true;
464  goto cleanup;
465  }
466 
467  worker.ReportProgress((int) (((double) y / instance.Header.NRows) * 100));
468  }
469  break;
470 
471  case Header.LayoutEnum.BSQ:
472  for (int b = 0; b < instance.Header.NBands; b++) {
473  ums[b].Write(buffer,
474  (int) (b * instance.Header.NRows * instance.Header.NCols),
475  (int) (instance.Header.NCols * instance.Header.NRows)
476  );
477 
478  if (worker.CancellationPending) {
479  e.Cancel = true;
480  goto cleanup;
481  }
482 
483  worker.ReportProgress((int) ((double) b / instance.Header.NBands));
484  }
485 
486  break;
487  }
488 
489  }
490  }
491  } catch (Exception ex) {
492  // Throw if anything unexpected happens.
493  throw new ParseException(ex.Message);
494  }
495 
496  cleanup:
497  // Unlock bitmaps.
498  instance.Bands.ForEach(b => b.UnlockBits());
499  e.Result = instance;
500 
501  // Stop timekeeping and log.
502  st.Stop();
503  MessageLog.Add(String.Format("Parsing took {0} msecs.", st.ElapsedMilliseconds));
504  }
505  }
