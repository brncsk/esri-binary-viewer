234  protected override void OnMouseMove(MouseEventArgs e) {
235  if (Image == null) {
236  return;
237  }
238 
239  CurrentImagePosition = new PointF(
240  (e.X - translation.X) / ZoomFactors[currentZoom],
241  (e.Y - translation.Y) / ZoomFactors[currentZoom]
242  );
243 
244  // Update the current pan based on the drag amount and the current zoom.
245  if (e.Button.HasFlag(MouseButtons.Left) &&
246  ((Image.Width * ZoomFactors[currentZoom] > Width) ||
247  (Image.Height * ZoomFactors[currentZoom] > Height))
248  ) {
249  translation = new PointF(e.X - mouseDragOrigin.X, e.Y - mouseDragOrigin.Y);
250 
251  if (Image.Width * ZoomFactors[currentZoom] < Width) {
252  translation.X = (Width / 2) - (ZoomFactors[currentZoom] * Image.Width / 2);
253  } else if (translation.X > 0) {
254  translation.X = 0;
255  } else if (translation.X < Width - (Image.Width * ZoomFactors[currentZoom])) {
256  translation.X = Width - (Image.Width * ZoomFactors[currentZoom]);
257  }
258 
259  if (Image.Height * ZoomFactors[currentZoom] < Height) {
260  translation.Y = (Height / 2) - (ZoomFactors[currentZoom] * Image.Height / 2);
261  } else if (translation.Y > 0) {
262  translation.Y = 0;
263  } else if (translation.Y < Height - (Image.Height * ZoomFactors[currentZoom])) {
264  translation.Y = Height - (Image.Height * ZoomFactors[currentZoom]);
265  }
266 
267  panning = true;
268  this.Refresh();
269  }
270 
271  base.OnMouseMove(e);
272  }

288  protected override void OnMouseWheel(MouseEventArgs e) {

293  (e.X - translation.X) / ZoomFactors[currentZoom],
294  (e.Y - translation.Y) / ZoomFactors[currentZoom]
295  );
296 
297  // Update zoom based on the current mouse position and zoom.
298  currentZoom += ((e.Delta > 0) ? +1 : -1);
299  if (currentZoom > (ZoomFactors.Length - 1)) {
300  currentZoom = ZoomFactors.Length - 1;
301  } else if (currentZoom < 0) {
302  currentZoom = 0;
303  }
304 
305  if (Image.Width * ZoomFactors[currentZoom] > Width)
306  {
307  translation.X = e.X - ZoomFactors[currentZoom] * imagePos.X;
308  } else {
309  translation.X = (Width / 2) - (ZoomFactors[currentZoom] * Image.Width / 2);
310  }
311 
312  if (Image.Height * ZoomFactors[currentZoom] > Height)
313  {
314  translation.Y = e.Y - ZoomFactors[currentZoom] * imagePos.Y;
315  } else {
316  translation.Y = (Height / 2) - (ZoomFactors[currentZoom] * Image.Height / 2);
