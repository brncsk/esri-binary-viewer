322  private void addBackgroundOperation(Tuple<BackgroundWorker, object> bw, string message, RunWorkerCompletedEventHandler rwceh) {
323  // We cancel any currently running operations before starting a new one.
324  // This might seem pointless but we don't actually have any operations that can be run in parallel.
325  // The reason for having async operations is not parallelism, but that we don't want to perform resource-heavy
326  // operations on the UI thread. We also want async ops to report their progress and be cancellable by the user
327  // if they're taking too much time.
328  if (currentBackgroundOperation != null) {
329  currentBackgroundOperation.CancelAsync();
330  }
331 
332  EventHandler cancelEventHandler = (o, a) => { currentBackgroundOperation.CancelAsync(); };
333 
334  // Set up event handlers on the BackgroundWorker.
335  currentBackgroundOperation = bw.Item1;
336  currentBackgroundOperation.ProgressChanged += (ps, pe) => { pgbAsyncOpProgress.Value = pe.ProgressPercentage; };
337  currentBackgroundOperation.RunWorkerCompleted += (o, a) => {
338  lblAsyncOpInfo.Text = "";
339  lblAsyncOpInfo.Click -= cancelEventHandler;
340  lblAsyncOpInfo.Visible = false;
341  pgbAsyncOpProgress.Visible = false;
342  lblFileInfo.Visible = true;
343  rwceh(o, a);
344  };
345 
346  // Set up UI.
347  lblFileInfo.Visible = false;
348  lblAsyncOpInfo.Visible = true;
349  lblAsyncOpInfo.Text = String.Format("{0} - {1}", message, "Click here to cancel operation...");
350  lblAsyncOpInfo.Click += cancelEventHandler;
351 
352  pgbAsyncOpProgress.Value = 0;
353  pgbAsyncOpProgress.Visible = true;
354 
355  // Enable progress reporting and cancellation, then start the BackgroundWorker.
356  currentBackgroundOperation.WorkerReportsProgress = true;
357  currentBackgroundOperation.WorkerSupportsCancellation = true;
358  currentBackgroundOperation.RunWorkerAsync(bw.Item2);
359  }
