412  private static void applyConvolutionFilterAsync (object sender, DoWorkEventArgs e) {
413  MessageLog.Add("Applying convolution filter.");
414  Stopwatch st = new Stopwatch();
415  st.Start();
416 
417  BackgroundWorker worker = sender as BackgroundWorker;
418  ConvolutionFilterArguments args = (ConvolutionFilterArguments) e.Argument;
419  Bitmap ob = new Bitmap(args.b);
420  BitmapData bd = args.b.LockBits(
421  new Rectangle(new Point(0, 0), args.b.Size),
422  ImageLockMode.ReadOnly,
423  PixelFormat.Format24bppRgb
424  );
425  BitmapData obd = ob.LockBits(
426  new Rectangle(new Point(0, 0), ob.Size),
427  ImageLockMode.WriteOnly,
428  PixelFormat.Format24bppRgb
429  );
430 
431 
432  // Assume a square kernel with an odd size
433  int kernelSize = (int) Math.Sqrt(args.kernel.Length);
434  int kernelHalf = (int) Math.Floor(kernelSize / 2d);
435  double kernelSum = args.kernel.Sum();
436 
437  unsafe {
438  byte* data = (byte*) bd.Scan0;
439  byte* odata = (byte*) obd.Scan0;
440  double[] sum = new double[3] { 0, 0, 0 };
441 
442  for (int y = kernelSize - 1; y < bd.Height - kernelSize; y++) {
443  for (int x = kernelSize - 1; x < bd.Width - kernelSize; x++) {
444  sum[0] = sum[1] = sum[2] = 0;
445 
446  for (int i = 0; i < kernelSize; i++) {
447  for (int j = 0; j < kernelSize; j++) {
448  int c = (y + i - kernelHalf) * bd.Stride + (x + j - kernelHalf) * 3;
449 
450  sum[0] += data[c] * args.kernel[i * kernelSize + j];
451  sum[1] += data[c + 1] * args.kernel[i * kernelSize + j];
452  sum[2] += data[c + 2] * args.kernel[i * kernelSize + j];
453  }
454  }
455 
456  sum[0] /= kernelSum;
457  sum[1] /= kernelSum;
458  sum[2] /= kernelSum;
459 
460  odata[y * bd.Stride + x * 3 + 0] = (byte)
461  ((sum[0] >= 0) ? (sum[0] <= 255 ? sum[0] : 255) : 0);
462  odata[y * bd.Stride + x * 3 + 1] = (byte)
463  ((sum[1] >= 0) ? (sum[1] <= 255 ? sum[1] : 255) : 0);
464  odata[y * bd.Stride + x * 3 + 2] = (byte)
465  ((sum[2] >= 0) ? (sum[2] <= 255 ? sum[2] : 255) : 0);
466  }
467 
468  if (worker.CancellationPending) {
469  e.Cancel = true;
470  goto cleanup;
471  }
472  worker.ReportProgress((int) ((double) y * 100 / bd.Height));
473  }
474  }
475 
476  cleanup:
477 
478  args.b.UnlockBits(bd);
479  ob.UnlockBits(obd);
480  e.Result = ob;
481  st.Stop();
482  MessageLog.Add(String.Format("Filter application took {0} msecs.", st.ElapsedMilliseconds));
483 
484  }
